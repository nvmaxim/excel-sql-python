(.*?);(.*?);(\d+);(\d+);(.*)
"$1";"$2";$3;$4;"$5"

Для manager:
Find: ^"([^"]+)".*$
Replace: "$1",

Для department:
Find: ^[^,]+, *"([^"]+)".*$
Replace: "$1"

Для revenue:
Find: ^[^,]+, *[^,]+, *([^,]+).*$
Replace: $1

Для experience:
Find: ^[^,]+, *[^,]+, *[^,]+, *([^,]+).*$
Replace: $1

Для period:
Find: ^.*"([^"]+)"$
Replace: "$1"

^ - начало строки

$ - конец строки

.* - любое количество любых символов

[^"] - любой символ, КРОМЕ кавычки

+ - один или больше раз

() - группа захвата (то, что попадает в $1)

* - ноль или больше пробелов
